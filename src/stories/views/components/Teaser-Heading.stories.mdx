import {
  ArgsTable,
  Meta,
  Story,
  Canvas,
  Preview,
} from "@storybook/addon-docs/blocks";

import teaserHeading from "./Teaser-Heading.hbs";

<Meta
  title="Komponenten/TeaserHeading"
  argTypes={{
    topline: {
      control: "text",
      description: "Topline Text",
    },
    headline: {
      control: "text",
      description: "Headline Text",
    },
    headlineTag: {
      control: {
        type: "select",
        options: ["span","h1", "h2", "h3", "h4", "h5", "h6"],
      },
      description: "Größe",
      table: {
        defaultValue: {
          summary: "span",
        },
      },
    },
    font: {
      description: "Schriftart",
      control: {
        type: 'inline-radio',
        options: ['headingSerif','heading']
      },
      table: {
        defaultValue: {
          summary: "heading",
        },
      },
    },
    size: {
      options: ['lg', 'sm'],
      control: { type: 'inline-radio' }
    },
  }}
/>

export const Template = ({ topline, ...args }) => {
  // You can either use a function to create DOM elements or use a plain html string!
  // return `<div>${label}</div>`;
  return teaserHeading({ topline, ...args }); 
};
